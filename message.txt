// ==========================================
// BEAST 感官捕獲系統 - 藍牙穩定版 (Bluetooth Serial)
// ==========================================

#include "BluetoothSerial.h"
#include "DHT.h"

// --- 藍牙設定 ---
#if !defined(CONFIG_BT_ENABLED) || !defined(CONFIG_BLUEDROID_ENABLED)
#error 你的 ESP32 板子沒有開啟藍牙功能！
#endif

BluetoothSerial SerialBT; // 建立藍牙物件

// --- 硬體腳位設定 ---
const int btnRecordPin = 0;   // 按鈕 A：內建 BOOT 鍵
const int btnSelectPin = 4;   // 按鈕 B：外接杜邦線切換餐廳
const int ledPin = 2;         // 內建藍色 LED

#define DHTPIN 15             // DHT11 接在 GPIO 15
#define DHTTYPE DHT11         
DHT dht(DHTPIN, DHTTYPE);

// --- 餐廳資料庫 ---
const int NUM_RESTAURANTS = 3;
int restIds[NUM_RESTAURANTS] = {101, 102, 103};
String restNames[NUM_RESTAURANTS] = {"麥當勞 (學餐)", "星巴克 (校門口)", "圖書館咖啡"};
int currentRestIndex = 0;

// --- 狀態與數據變數 ---
bool isRecording = false;
unsigned long lastSampleTime = 0;
const unsigned long sampleInterval = 2000;

float totalNoise = 0, totalTemp = 0, totalHumid = 0;
int sampleCount = 0;
unsigned long startTime = 0;

void setup() {
  Serial.begin(115200);
  
  pinMode(btnRecordPin, INPUT_PULLUP);
  pinMode(btnSelectPin, INPUT_PULLUP);
  pinMode(ledPin, OUTPUT);
  digitalWrite(ledPin, LOW); 
  dht.begin();

  // 啟動藍牙，設定藍牙名稱！(這是你在電腦上會看到的裝置名稱)
  SerialBT.begin("BEAST_ESP32_IoT"); 
  
  Serial.println("\n=================================");
  Serial.println("✅ 藍牙已啟動！");
  Serial.println("👉 請在電腦端開啟藍牙，並配對「BEAST_ESP32_IoT」");
  Serial.println("=================================\n");
  printCurrentRestaurant();
}

void loop() {
  // 切換餐廳
  if (digitalRead(btnSelectPin) == LOW && !isRecording) {
    delay(50);
    if (digitalRead(btnSelectPin) == LOW) {
      currentRestIndex = (currentRestIndex + 1) % NUM_RESTAURANTS;
      printCurrentRestaurant();
      while(digitalRead(btnSelectPin) == LOW) { delay(10); }
    }
  }

  // 開始 / 結束錄製
  if (digitalRead(btnRecordPin) == LOW) {
    delay(50);
    if (digitalRead(btnRecordPin) == LOW) {
      if (!isRecording) {
        startCapture();
      } else {
        stopCapture();
      }
      while(digitalRead(btnRecordPin) == LOW) { delay(10); }
    }
  }

  // 採集數據
  if (isRecording && (millis() - lastSampleTime >= sampleInterval)) {
    collectMixedSensorData();
    lastSampleTime = millis();
  }
}

void printCurrentRestaurant() {
  Serial.print("📍 鎖定地點: ");
  Serial.println(restNames[currentRestIndex]);
}

void startCapture() {
  isRecording = true;
  digitalWrite(ledPin, HIGH); 
  sampleCount = 0; totalNoise = 0; totalTemp = 0; totalHumid = 0;
  startTime = millis();
  Serial.println("\n[事件] 👉 開始錄製 (藍牙版)！");
}

void collectMixedSensorData() {
  float currentTemp = dht.readTemperature();
  float currentHumid = dht.readHumidity();
  
  if (isnan(currentTemp) || isnan(currentHumid)) {
    currentTemp = 25.0; currentHumid = 60.0;
  }
  float currentNoiseDb = random(500, 800) / 10.0;

  totalTemp += currentTemp;
  totalHumid += currentHumid;
  totalNoise += currentNoiseDb;
  sampleCount++;
  Serial.printf("   ⚙️ 採集中... 溫度: %.1f°C | 濕度: %.1f%%\n", currentTemp, currentHumid);
}

void blinkSuccessLED() {
  for (int i = 0; i < 3; i++) {
    digitalWrite(ledPin, HIGH);
    delay(80);   // 亮 80 毫秒
    digitalWrite(ledPin, LOW);
    delay(80);   // 暗 80 毫秒
  }
}

void stopCapture() {
  isRecording = false;
  // 先不要關閉 LED，留給後面的閃爍動畫用

  float avgTemp = (sampleCount > 0) ? (totalTemp / sampleCount) : 0;
  float avgHumid = (sampleCount > 0) ? (totalHumid / sampleCount) : 0;
  float avgNoise = (sampleCount > 0) ? (totalNoise / sampleCount) : 0;
  int durationSec = (millis() - startTime) / 1000;

  // 組裝 JSON
  String jsonPayload = "{\"restaurant_id\": " + String(restIds[currentRestIndex]) + 
                       ", \"avg_noise_db\": " + String(avgNoise, 1) + 
                       ", \"avg_temp_c\": " + String(avgTemp, 1) + 
                       ", \"avg_humidity\": " + String(avgHumid, 1) + 
                       ", \"duration_sec\": " + String(durationSec) + "}";

  // 👉 1. 這裡會印在 Arduino IDE 的序列埠監控視窗 (USB)
  Serial.println("\n[事件] 👈 結束錄製！");
  Serial.println("📤 透過藍牙發送 JSON: " + jsonPayload);
  
  // 👉 2. 這裡會把資料透過藍牙丟給 Python 後端
  SerialBT.println(jsonPayload); 
  
  // 👉 3. 呼叫我們剛剛寫的動畫：快速閃爍 3 下代表發送完成！
  blinkSuccessLED(); 
  
  // 確保最後燈是關著的，等待下一次錄製
  digitalWrite(ledPin, LOW); 

  Serial.println("✅ 藍牙傳輸完成！");
  Serial.println("--------------------------------------------------");
}
